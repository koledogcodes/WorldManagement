package me.koledogcodes.worldcontrol.configs;

import java.io.File;
import java.io.IOException;
import java.util.HashMap;

import org.bukkit.configuration.file.FileConfiguration;
import org.bukkit.configuration.file.YamlConfiguration;

public class PlayerDataFile {
	
	private String dataFolder = "plugins/WorldManagement/";
	private HashMap<String, File> file = new HashMap<String, File>();

	public PlayerDataFile(String player) {
		file.put(player, new File(dataFolder + "Userdata", player + ".yml"));
		generateConfig(player);
	}
	
	private void generateConfig(String player){
		if (file.get(player) != null){
			return;
		}
		else {
			try {
				file.get(player).createNewFile();
			} catch (IOException e) {
				// TODO Auto-generated catch block
				e.printStackTrace();
			}
		}
	}
	
	public FileConfiguration getConfig(String player){
		return YamlConfiguration.loadConfiguration(file.get(player));
	}
	
	public void saveConfig(String player){
		try {
			getConfig(player).save(file.get(player));
		} catch (IOException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}
	
	public void reloadConfig(String player){
		generateConfig(player);
		saveConfig(player);
	}
}
